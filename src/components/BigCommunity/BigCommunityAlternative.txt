import { useState } from 'react'
import ShowComponent from '../ShowComponent/ShowComponent';

const BigCommunity = () => {
  const [isSectionVisible, setIsSectionVisible] = useState(true);

  const toggleSectionVisibility = (data) => {
    setIsSectionVisible(data);
  };

  return (
    <div>
      <ShowComponent sendDataToParent={toggleSectionVisibility} isSectionVisible={isSectionVisible}/>
      {isSectionVisible && <p>Hola</p>}
    </div>
  );
}

export default BigCommunity
